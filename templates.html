async function fetchStockData() {

  const stockTableBody = document.getElementById("stockTableBody");

  stockTableBody.innerHTML = ""; // Clear the table before updating


  try {

    const response = await fetch("/stocks?symbols=AAPL,GOOGL,AMZN,MSFT");

    const stockData = await response.json();


    for (const stock of stockData) {

      const row = `

        <tr>

          <td>${stock.symbol}</td>

          <td>$${stock.price.toFixed(2)}</td>

          <td class="${stock.change >= 0 ? 'text-success' : 'text-danger'}">${stock.change.toFixed(2)}</td>

        </tr>

      `;

      stockTableBody.innerHTML += row;

    }

  } catch (error) {

    console.error("Failed to fetch stock data:", error);

  }

}
